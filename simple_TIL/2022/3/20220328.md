export const deleteBookmark = function (id) {
  // Delete bookmark
  const index = state.bookmarks.findIndex((el) => el.id === id);

[N, ...arr] = require("fs")
  .readFileSync(process.platform == "linux" ? "/dev/stdin" : "예제.txt")
  .toString()
  .trim()
  .split("\n");

dp = new Array(N + 1).fill(0);

for (let i = N - 1; i > -1; i--) {
  if (i + arr[i][0] > N) {
    dp[i] = dp[i + 1];
  } else {
    // 해당일의 일일을 했을때 가치 vs 해당일의 일을 하지 않았을때 가치

function solution(record) {
  var answer = [];
  let methodList = {
    Enter: "들어왔습니다",
    Leave: "나갔습니다",
    Change: false,
  };
  let Users = new Map();
  record.map((el) => {
    const [method, id, nickName] = el.split(" ");
    // leave는 닉네임이 없으니 바꾸면 ㄴㄴ
    nickName && Users.set(id, nickName);
  });

  record.map((el) => {
    const [method, id, nickName] = el.split(" ");
    // change는 로그에 남길필요가 없다
    methodList[method] &&
      answer.push(`${Users.get(id)}님이 ${methodList[method]}.`);
  });
  return answer;
}
