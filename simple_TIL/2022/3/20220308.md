function solution(s) {
  var answer = 0;
  let table = [
    "zero",
    "one",
    "two",
    "three",
    "four",
    "five",
    "six",
    "seven",
    "eight",
    "nine",
  ];
  while (/[a-z-A-Z]/g.test(s)) {
    table.map((el, idx) => {
      s = s.replace(el, idx);
    });
  }
  answer = Number(s);
  return answer;
}
